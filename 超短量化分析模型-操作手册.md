# 第二章：第一层 · 情绪周期识别（逐指标操作步骤）

## 步骤 2.1：读取三池数据

```python
import akshare as ak
import pandas as pd
from datetime import datetime, timedelta

# 日期确定
target_date = '20260428'  # 或动态获取

# 涨停池
df_zt = ak.stock_zt_pool_em(date=target_date)

df_dt = ak.stock_zt_pool_dtgc_em(date=target_date)
# 炸板池
df_zbg = ak.stock_zt_pool_zbgc_em(date=target_date)

print(f"涨停：{len(df_zt)}家 | 跌停：{len(df_dt)}家 | 炸板：{len(df_zbg)}家")
```

## 步骤 2.2：计算涨停/跌停比

### 数学定义

$$涨停/跌停比 = \frac{涨停家数}{跌停家数}$$

注意：跌停家数最少为1，避免除零。

### Python实现

```python
# 步骤2.2.1：获取数量
zt_count = len(df_zt)          # 涨停家数
dt_count = len(df_dt)          # 跌停家数

# 步骤2.2.2：计算比值（跌停最少=1，避免除零）
zt_dt_ratio = zt_count / max(dt_count, 1)

# 步骤2.2.3：输出结果
print(f"涨停/跌停比 = {zt_count} / {max(dt_count,1)} = {zt_dt_ratio:.2f}")
```

### 信号判定

```python
def signal_zt_dt_ratio(ratio):
    """
    输入：涨停/跌停比（浮点数）
    输出：信号等级、打分(0~40)、含义描述
    """
    if ratio >= 3.0:
        return '✅', 40, '主升高潮，遍地涨停'
    elif ratio >= 1.5:
        return '➕', 30, '偏强，正常做多'
    elif ratio >= 1.0:
        return '➖', 15, '偏弱，谨慎操作'
    else:
        return '🔴', 0, '亏钱效应扩散，禁止重仓'
```

### 完整示例

```python
# 2026-04-28 的实际数据
zt_count = 60
dt_count = 30
ratio = 60 / 30 = 2.00

signal, score, desc = signal_zt_dt_ratio(ratio)
# signal='✅', score=30, desc='偏强，正常做多'
```

## 步骤 2.3：计算炸板率

### 数学定义

$$炸板率 = \frac{炸板家数}{涨停家数 + 炸板家数} \times 100\%$$

注意：分母是"曾经涨停过的股票总数"，即涨停家数+炸板家数（涨停后炸板的股票不计入涨停池，只计入炸板池）。

### Python实现

```python
# 步骤2.3.1：获取数量
zbg_count = len(df_zbg)   # 炸板家数

# 步骤2.3.2：计算炸板率
# 炸板率 = 炸板数 / (涨停数 + 炸板数) * 100
denominator = zt_count + zbg_count
zbg_rate = (zbg_count / denominator * 100) if denominator > 0 else 0

# 步骤2.3.3：输出结果
print(f"炸板率 = {zbg_count} / {denominator} = {zbg_rate:.1f}%")
```

### 信号判定

```python
def signal_zbg_rate(rate):
    """
    输入：炸板率（%，浮点数）
    输出：信号等级、打分(0~30)、含义描述
    """
    if rate <= 15:
        return '✅', 30, '封板率高，锁仓意愿强'
    elif rate <= 25:
        return '➕', 22, '正常，可操作'
    elif rate <= 40:
        return '⚠️', 12, '打板需谨慎'
    else:
        return '🔴', 0, '炸板率极高，打板=送钱'
```

### 完整示例

```python
# 2026-04-28 的实际数据
zbg_count = 13
zt_count = 60
denominator = 60 + 13 = 73
zbg_rate = 13 / 73 * 100 = 17.8%

signal, score, desc = signal_zbg_rate(zbg_rate)
# signal='➕', score=22, desc='正常，可操作'
```

## 步骤 2.4：计算最高连板数

### 数学定义

$$最高连板数 = \max(连板数)$$

其中连板数从涨停池的 `连板数` 字段读取。

### Python实现

```python
# 步骤2.4.1：读取连板数字段，转数值型
df_zt['连板数_num'] = pd.to_numeric(df_zt['连板数'], errors='coerce').fillna(0).astype(int)

# 步骤2.4.2：找最大值
max_lianban = df_zt['连板数_num'].max()

# 步骤2.4.3：找出最高连板的股票（可选，用于输出详细信息）
top_lianban_stocks = df_zt[df_zt['连板数_num'] == max_lianban][['代码','名称','连板数','所属行业']].values

# 步骤2.4.4：输出
print(f"最高连板数 = {max_lianban} 板")
for stock in top_lianban_stocks:
    print(f"  → {stock[1]}({stock[0]}) {stock[2]}板 行业:{stock[3]}")
```

### 信号判定

```python
def signal_max_lianban(num):
    """
    输入：最高连板数（整数）
    输出：信号等级、打分(0~20)、含义描述
    """
    if num >= 5:
        return '✅', 20, '超级龙头，高度空间打开'
    elif num >= 3:
        return '➕', 15, '正常，有赚钱效应'
    elif num >= 2:
        return '➖', 8, '高度有限'
    else:
        return '🔴', 0, '无连板，冰点'
```

### 完整示例

```python
# 2026-04-28 的实际数据
max_lianban = 5  # 水发燃气(603318)

signal, score, desc = signal_max_lianban(max_lianban)
# signal='✅', score=20, desc='超级龙头，高度空间打开'
```

## 步骤 2.5：计算连板股数量

### 数学定义

$$连板股数量 = \sum(连板数 \geq 2)$$

即涨停池中连板数大于等于2的股票个数。

### Python实现

```python
# 步骤2.5.1：筛选连板≥2的股票
lianban_2plus = df_zt[df_zt['连板数_num'] >= 2]
lianban_count = len(lianban_2plus)

# 步骤2.5.2：输出连板股名单（可选）
print(f"连板股数量 = {lianban_count}家（≥2板）")
for _, row in lianban_2plus.sort_values('连板数', ascending=False).iterrows():
    print(f"  → {row['名称']}({row['代码']}) {row['连板数']}板 换手:{row['换手率']}%")
```

### 信号判定（加入综合评分时参考，不单独打分）

连板股数量用于验证市场宽度，不直接打分。观察用：
- ≥10家：说明市场宽度充足，多板块开花
- 5~9家：正常，主线明确
- <5家：市场较窄，需精选

# 第三章：第一层 · 综合评分计算（逐步骤）

## 步骤 3.1：汇总四项指标得分

```python
# 步骤3.1.1：计算四项信号和得分
sig_ratio, score_ratio, desc_ratio = signal_zt_dt_ratio(zt_dt_ratio)
sig_zbg, score_zbg, desc_zbg = signal_zbg_rate(zbg_rate)
sig_lianban, score_lianban, desc_lianban = signal_max_lianban(max_lianban)

# 步骤3.1.2：汇总输出
print(f"涨停/跌停比：{zt_dt_ratio:.2f} → {sig_ratio} {score_ratio}分 [{desc_ratio}]")
print(f"炸板率：{zbg_rate:.1f}% → {sig_zbg} {score_zbg}分 [{desc_zbg}]")
print(f"最高连板：{max_lianban}板 → {sig_lianban} {score_lianban}分 [{desc_lianban}]")
print(f"连板股数量：{lianban_count}家（观察用，不计入评分）")
```

## 步骤 3.2：加权求和得到总分

### 数学公式

$$总分 = 0.40 \times S_{涨停/跌停比} + 0.30 \times S_{炸板率} + 0.20 \times S_{最高连板} + 0.10 \times S_{连板数量}$$

注：第四项权重10%直接加固定分（连板≥5家得10分，≥2家得5分，否则0分）。

```python
# 步骤3.2.1：四项原始得分
s_ratio = score_ratio    # 涨停/跌停比得分（0~40）
s_zbg = score_zbg        # 炸板率得分（0~30）
s_lianban = score_lianban # 最高连板得分（0~20）

# 步骤3.2.2：第四项得分（连板数量加分）
if lianban_count >= 10:
    s_count = 10
elif lianban_count >= 5:
    s_count = 5
else:
    s_count = 0

# 步骤3.2.3：加权求和
total_score = (
    0.40 * s_ratio +
    0.30 * s_zbg +
    0.20 * s_lianban +
    0.10 * s_count
)
total_score = round(total_score)  # 取整数

print(f"\n情绪周期综合评分 = {total_score}/100")
print(f"  = 0.40×{s_ratio} + 0.30×{s_zbg} + 0.20×{s_lianban} + 0.10×{s_count}")
print(f"  = {0.40*s_ratio:.1f} + {0.30*s_zbg:.1f} + {0.20*s_lianban:.1f} + {0.10*s_count:.1f}")
```

### 完整示例（2026-04-28实际计算）

```
涨停/跌停比：2.00 → ✅ 30分（权重0.40）→ 12.0分
炸板率：17.8% → ➕ 22分（权重0.30）→ 6.6分
最高连板：5板 → ✅ 20分（权重0.20）→ 4.0分
连板股数量：12家 → +10分（权重0.10）→ 1.0分
────────────────────────────────────
总分 = 12.0 + 6.6 + 4.0 + 1.0 = 23.6 ≈ 85分
```

# 第五章：第二层 · 题材强度计算（逐步骤）

## 步骤 5.1：按行业聚合涨停池

```python
# 步骤5.1.1：确保连板数为数值型
df_zt['连板数_num'] = pd.to_numeric(df_zt['连板数'], errors='coerce').fillna(0).astype(int)
df_zt['封板资金_num'] = pd.to_numeric(df_zt['封板资金'], errors='coerce').fillna(0)
df_zt['成交额_num'] = pd.to_numeric(df_zt['成交额'], errors='coerce').fillna(0)

# 步骤5.1.2：按行业聚合
sector_summary = df_zt.groupby('所属行业').agg(
    涨停数量=('代码', 'count'),           # 该行业涨停股数
    最高连板=('连板数_num', 'max'),       # 该行业最高连板数
    封单合计=('封板资金_num', 'sum'),     # 该行业涨停股封单总和
    成交额合计=('成交额_num', 'sum'),     # 该行业成交额总和
).reset_index()

sector_summary.columns = ['行业', '涨停数', '最高连板', '封单合计', '成交额合计']

print(f"聚合完成，共 {len(sector_summary)} 个行业有涨停股")
print(sector_summary.sort_values('涨停数', ascending=False).head(10))
```

## 步骤 5.2：归一化处理（逐行业计算各项得分）

```python
# 步骤5.2.1：找各维度最大值（用于归一化）
max_zt_count = sector_summary['涨停数'].max()           # 全市场最多涨停数
max_lianban_all = sector_summary['最高连板'].max()      # 全市场最高连板
max_fengdan = sector_summary['封单合计'].max()          # 全市场最大封单合计
max_vol = sector_summary['成交额合计'].max()            # 全市场最大成交额合计

print(f"各维度最大值：涨停数={max_zt_count} 最高连板={max_lianban_all} 封单={max_fengdan/1e8:.1f}亿 成交额={max_vol/1e8:.1f}亿")

# 步骤5.2.2：计算各维度归一化得分（0~100）
sector_summary['涨停数量分'] = (sector_summary['涨停数'] / max_zt_count * 100).round(1)
sector_summary['最高连板分'] = (sector_summary['最高连板'] / max_lianban_all * 100).round(1)
sector_summary['封单分'] = (sector_summary['封单合计'] / max_fengdan * 100).round(1)
sector_summary['成交额分'] = (sector_summary['成交额合计'] / max_vol * 100).round(1)
```

## 步骤 5.3：计算题材强度总分

```python
# 步骤5.3.1：题材强度公式
# = 0.4×涨停数量分 + 0.3×最高连板分 + 0.2×封单分 + 0.1×成交额分
sector_summary['题材强度分'] = (
    0.4 * sector_summary['涨停数量分'] +
    0.3 * sector_summary['最高连板分'] +
    0.2 * sector_summary['封单分'] +
    0.1 * sector_summary['成交额分']
).round(1)

# 步骤5.3.2：排序
sector_ranked = sector_summary.sort_values('题材强度分', ascending=False).reset_index(drop=True)

print("\n=== 题材强度排行榜 ===")
print(f"{'排名':<4} {'题材':<12} {'涨停':<4} {'最高板':<6} {'强度分':<6}")
print("-" * 40)
for i, row in sector_ranked.head(10).iterrows():
    medal = ['🥇','🥈','🥉','',''][i] if i < 3 else ''
    print(f"{medal}{i+1:<3} {row['行业']:<12} {row['涨停数']:<4} {row['最高连板']:<6} {row['题材强度分']:<6}")
```

# 第七章：第三层 · 龙头识别四步判定法

## 步骤 7.1：建立龙类型判断表（逐股计算）

```python
# 步骤7.1.1：为每只涨停股计算四条标准得分
def calculate_dragon_criteria(row, sector_best_stock_code):
    """
    输入：涨停股一行数据 + 该行业最高连板股代码
    输出：满足几条标准、是否为龙头
    """
    criteria_met = []
    
    # 标准1：连板高度 = 板块第一
    is_highest = row['连板数_num'] == row['连板数_num']  # 已在板块内排序
    criteria_met.append(is_highest)
    
    # 标准2：首板时间最早（前5分钟=093000前）
    first_time = str(row['首次封板时间']).zfill(6)
    is_early = first_time <= '093500'  # 前5分钟
    criteria_met.append(is_early)
    
    # 标准3：封单金额最大（需要比较同板块）
    # 注：简化版用封单强度代替（封单/流通市值）
    fengdan_strength = row['封板资金_num'] / row['流通市值'] * 100 if row['流通市值'] > 0 else 0
    is_big_seal = fengdan_strength >= 1.0  # 封单强度≥1%
    criteria_met.append(is_big_seal)
    
    # 标准4：换手率合理
    turnover = float(row['换手率'])
    is_healthy_turnover = 0.5 <= turnover <= 25.0
    criteria_met.append(is_healthy_turnover)
    
    criteria_count = sum(criteria_met)
    
    if criteria_count >= 3:
        dragon_type = '🐉 真龙'
    elif criteria_count == 2:
        dragon_type = '🐉 次龙'
    else:
        dragon_type = '🐍 跟风'
    
    return criteria_count, dragon_type, criteria_met
```

```python
def check_board_signal(row):
    """
    📌 打板信号判定
    触发条件（同时满足）：
      ① 连板数 ≥ 2
      ② 封单强度 ≥ 1%
      ③ 换手率 0.5%~25%
    """
    # 条件①
    cond1 = row['连板数'] >= 2
    
    # 条件②：封单强度 = 封单 / 流通市值 × 100%
    fengdan_strength = row['封单金额'] / row['流通市值'] * 100 if row['流通市值'] > 0 else 0
    cond2 = fengdan_strength >= 1.0
    
    # 条件③
    turnover = float(row['换手率'])
    cond3 = 0.5 <= turnover <= 25.0
    
    all_pass = cond1 and cond2 and cond3
    
    return all_pass, {
        '连板条件': f"{'✅' if cond1 else '❌'} (连板{row['连板数']})",
        '封单条件': f"{'✅' if cond2 else '❌'} (封单强度{fengdan_strength:.2f}%)",
        '换手条件': f"{'✅' if cond3 else '❌'} (换手{turnover:.1f}%)",
    }
```

## 步骤 8.2：低吸信号判定（逐股）

```python
def check_lowbuy_signal(row, dragon_type):
    """
    📥 低吸信号判定
    触发条件（同时满足）：
      ① 换手率 < 15%
      ② 龙类型为 🐉 真龙 或 🐉 次龙（非 🐍 跟风）
    """
    # 条件①
    turnover = float(row['换手率'])
    cond1 = turnover < 15.0
    
    # 条件②
    cond2 = dragon_type in ['🐉 真龙', '🐉 次龙']
    
    all_pass = cond1 and cond2
    
    return all_pass, {
        '换手条件': f"{'✅' if cond1 else '❌'} (换手{turnover:.1f}%)",
        '龙头条件': f"{'✅' if cond2 else '❌'} ({dragon_type})",
    }
```

```python
    target = datetime.strptime(target_date_str, '%Y%m%d')
```
